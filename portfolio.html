    # Mensagem de boas-vindas
print("Bem-vindo(a) ao sistema de pedidos da loja de Açaí e Cupuaçu!")
print("Desenvolvido por: Maria Eduarda Gonçalves Alves\n")

total_pedido = 0.0

print("Tabela de Preços - Loja de Açaí e Cupuaçu")
print("-" * 40)
print(f"{'Tamanho':<10} {'Cupuaçu (CP)':<15} {'Açaí (AC)':<10}")
print("-" * 40)
print(f"{'P':<10} R$ {9:<12} R$ {11}")
print(f"{'M':<10} R$ {14:<12} R$ {16}")
print(f"{'G':<10} R$ {18:<12} R$ {20}")
print("-" * 40)

while True:
    sabor = input("Digite o sabor desejado (CP para Cupuaçu ou AC para Açaí): ").upper()
    if sabor != "CP" and sabor != "AC":
        print("Sabor inválido. Tente novamente.\n")
        continue

    tamanho = input("Digite o tamanho desejado (P, M ou G): ").upper()
    if tamanho not in ["P", "M", "G"]:
        print("Tamanho inválido. Tente novamente.\n")
        continue

    if sabor == "CP":
        if tamanho == "P":
            preco = 9
        elif tamanho == "M":
            preco = 14
        else:
            preco = 18
    elif sabor == "AC":
        if tamanho == "P":
            preco = 11
        elif tamanho == "M":
            preco = 16
        else:
            preco = 20

    total_pedido += preco
    print(f"Pedido registrado: {'Cupuaçu' if sabor == 'CP' else 'Açaí'} tamanho {tamanho} - R$ {preco:.2f}\n")

    mais = input("Deseja pedir mais alguma coisa? (S/N): ").upper()
    if mais != "S":
        break

print(f"\nTotal do pedido: R$ {total_pedido:.2f}")
print("Obrigado pela sua compra!\n")
